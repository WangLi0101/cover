# 提取视频针

![封面图](../photo/video-frame-cover.webp)

#### 实现原理 利用canvas

```html
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>
  <body>
    <input type="file" />
    <script>
      // 提取视频针
      const captureFrame = (file, time) => {
        return new Promise((resolve) => {
          const url = URL.createObjectURL(file);
          const videoDocum = document.createElement("video");
          // 浏览器机制，需要静音才能自动播放
          videoDocum.muted = true;
          videoDocum.autoplay = true;
          videoDocum.src = url;
          videoDocum.currentTime = time;
          videoDocum.oncanplay = () => {
            const cvs = document.createElement("canvas");
            const ctx = cvs.getContext("2d");
            cvs.width = videoDocum.videoWidth;
            cvs.height = videoDocum.videoHeight;
            ctx.drawImage(videoDocum, 0, 0, cvs.width, cvs.height);
            cvs.toBlob((blob) => {
              resolve({
                url: URL.createObjectURL(blob),
              });
            });
          };
        });
      };

      const inputElement = document.querySelector("input");
      inputElement.addEventListener("change", (e) => {
        const file = e.target.files[0];
        for (let i = 0; i < 50; i++) {
          captureFrame(file, i).then((res) => {
            const img = document.createElement("img");
            img.src = res.url;
            document.body.appendChild(img);
          });
        }
      });
    </script>
  </body>
</html>

```
