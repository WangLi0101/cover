### 方法一

```javascript
function unique(arr) {
  return [...new Set(arr)];
}
```

### 方法二

```javascript
function unique(arr) {
  // 因为 indexOf 只会返回第一个匹配项的索引，所以只要当前索引和 indexOf 返回的索引不一致，就说明当前项不是第一个匹配项
  return arr.filter((item, index) => arr.indexOf(item) === index);
}
```

### 方法三

```javascript
function unique(arr) {
  return arr.reduce(
    (prev, cur) => (prev.includes(cur) ? prev : [...prev, cur]),
    []
  );
}
```

### 方法四

```javascript
function unique(arr) {
  return Object.keys(
    arr.reduce((a, b) => {
      return { ...a, [b]: true };
    }, {})
  );
}
```

> 如果要去重的数组元素是对象，可以使用 `JSON.stringify` 将对象序列化，然后再进行去重。
