#### 方法一 `Clipboard API`

> 只能在 https 下使用

```html
<input
  type="text"
  id="textToCopy"
  value="This text will be copied to clipboard"
/>
<button onclick="copyToClipboard()">Copy to Clipboard</button>
```

```javascript
function copyToClipboard() {
  // 获取输入框的内容
  const text = document.getElementById("textToCopy").value;

  // 使用 Clipboard API 写入剪贴板
  navigator.clipboard
    .writeText(text)
    .then(() => {
      alert("Text copied to clipboard!");
    })
    .catch((err) => {
      console.error("Failed to copy text: ", err);
    });
}
```

#### 方法二 `execCommand`

```html
<input
  type="text"
  id="textToCopy"
  value="This text will be copied using execCommand"
/>
<button onclick="copyToClipboard()">Copy to Clipboard</button>
```

```javascript
function copyToClipboard() {
  // 获取输入框的元素
  const textElement = document.getElementById("textToCopy");

  // 选择输入框的文本
  textElement.select();

  // 执行复制命令
  try {
    const successful = document.execCommand("copy");
    if (successful) {
      alert("Text copied to clipboard!");
    } else {
      alert("Unable to copy text.");
    }
  } catch (err) {
    console.error("Failed to copy text: ", err);
  }

  // 清除选择（可选步骤，视觉清理）
  window.getSelection().removeAllRanges();
}
```
