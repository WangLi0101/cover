Counter.tsx 
const Counter = () => {
  // 定义状态 [当前值, 更新函数]
  const [count, setCount] = useState(0);

  const increment = () => {
    // 推荐：使用回调函数更新
    setCount((prev) => prev + 1);
  };

  return (
    <button onClick={increment}>
      {count}
    </button>
  );
};
